## Подгрузить изменения из GIT
git pull
## Статус GIT
git status
## Посмотреть изменения GIT
git sync
## Коммит GIT
git add .
git commit -m "Some message"
git push
## or
git push --set-upstream origin {name_of_branch}
## Создать свой branch GIT
git checkout -b "grekhov/my_first_branch"
